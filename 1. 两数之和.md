## 两数之和

https://leetcode-cn.com/problems/two-sum/

## C#
```csharp
public class Solution
{
    public int[] TwoSum(int[] nums, int target)
    {
        var dic = new Dictionary<int, int>();
        var result = new int[2];

        for (var i = 0; i < nums.Length; i++)
        {
            if (dic.ContainsKey(nums[i]))
            {
                result[0] = dic[nums[i]]; 
                result[1] = i;
                return result;
            }

            dic[target - nums[i]] = i;
        }

        return result;
    }
}
```
